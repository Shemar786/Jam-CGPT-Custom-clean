        int i = 0;
        prepareNext();
        while (start-- > 0) {
            nextChar();
            i++;
        }
        int c;
        while ((c = nextChar()) != 0) {
            if (c == ch)
                return i;
            i++;
        }
        if (ch == 0)
            return i;
        return -1;
