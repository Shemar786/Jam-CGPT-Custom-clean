TDAT:     protected String getString(String key, Object... args) {
        String answer = null;
        try {
            answer = bundle.getString(makeUnique(key));
        }
        catch (MissingResourceException e) {
            return "!" + e.getKey() + "!";
        }

        if (0 != args.length) {
            return MessageFormat.format(answer, args);
        }

        return answer;
    }
