		GameData gameData = GameData.getInstance();
		switch (method) {
			case fromDeck:
				//pickup from deck
				addCard(gameData.getDeck().popTopCard());
				break;
			case fromThrown:
				//pickup from thrown
				addCard(gameData.getThrownCards().popTopCard());
				break;
			case decidePickup:
				//decide and add
				if (strategy!=null){
					addCard(strategy.decidePickUp());
				}else{
					addCard(gameData.getDeck().popTopCard());
				}
				break;
		}
	}
