TDAT: 	protected boolean validXML(File document) throws ExecutorException{
		boolean b = false;
		try{
			XMLReader reader = XMLReaderFactory.createXMLReader();
			reader.parse(document.getAbsolutePath());
			b=true;
		} catch (SAXException e) {
			b=false;
		} catch (IOException e) {
			throw new ExecutorException(e,ExecutorErrorCodes.EXECUTOR_READING_ERROR+" "+document.getName());
		}
		return b;
	}
