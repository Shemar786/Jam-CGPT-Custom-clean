TDAT:         public void showSearchField(String text) {
            if (StringUtils.isEmpty(text)) {
                remove(searchField);
            } else {
                searchField.setText(text);
                if (searchField.getParent() == null) {
                    add(searchField);
                }
            }
            searchField.invalidate();
            validate();
        }
