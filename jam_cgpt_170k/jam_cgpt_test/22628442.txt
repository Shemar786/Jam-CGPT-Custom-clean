		if ((list != null) && (s != null)) {
			for (int i = 0; i < list.size(); i++) {
				if (s.compareToIgnoreCase((String) list.get(i)) <= 0) {
					list.add(i, s);
					return;
				}
			}
			list.add(s);
		}
	}
