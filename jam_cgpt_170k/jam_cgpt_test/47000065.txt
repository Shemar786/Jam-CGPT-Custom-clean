TDAT:     private String skipSpaces(String str) {
        StringBuilder newstr = new StringBuilder(100);
        int len = str.length();

        for (int i = 0; i < len; i++) {
            if (str.charAt(i) != ' ') {
                newstr.append(str.charAt(i));
            }
        }

        return newstr.toString();
    }
