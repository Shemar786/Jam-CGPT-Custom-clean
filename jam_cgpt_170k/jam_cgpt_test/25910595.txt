			Uri.Builder builder = AssetProvider.CONTENT_URI.buildUpon();

			builder.appendQueryParameter(AssetProvider.PARAM_CONTENT_TYPE,

			Uri uri = builder.build();
			Intent intent = new Intent(Intent.ACTION_VIEW, uri);
