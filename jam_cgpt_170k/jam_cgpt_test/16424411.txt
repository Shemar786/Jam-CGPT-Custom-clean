		super.act(action, feedback);
		Structure actionTerm = action.getActionTerm();
		String functor = actionTerm.getFunctor();
		Method m = annotatedMethods.get(functor);
		if (m != null) {
			try {
				m.invoke(this, (Object)actionTerm.getTermsArray());
			} catch (Exception e) {
				e.printStackTrace();
			}
		}
	}
