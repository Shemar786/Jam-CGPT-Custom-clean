TDAT:     private VirtualDirectory getDirectory(String path) {
        VirtualDirectory result = this.fRoot;
        String[] tokens = PathUtil.getPathTokens(path);

        for (int i = 0; i < tokens.length; i++) {
            if (!tokens[i].equals("")) {
                if (null != result) {
                    result = result.getDirectory(tokens[i]);
                } else {
                    break;
                }
            }
        }

        return result;
    }
