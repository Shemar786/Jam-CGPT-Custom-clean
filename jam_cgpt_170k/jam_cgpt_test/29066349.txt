TDAT: 	public OlapModel getModel(String id){
		OlapModel olapModel=(OlapModel) models.get(id);
		if(olapModel==null){
			try {
				if(modeXmla){
					olapModel=createXmlaModel();
				}else{
					olapModel=createMondrianModel();
				}
				olapModel.setID(id);
				models.put(id, olapModel);
				OlapModelManagerEvent event
					=new OlapModelManagerEvent(this,olapModel);
				fireModelAdded(event);
				if(olapModel.getID().equals(DEFAULT_MODEL)){
					fireDefaultModelChanged(event);
				}
				logger.debug("create "+id+" OlapModel");				
			} catch (Exception e) {
				logger.error(e);
			}
		}
		return olapModel;
		
	}
