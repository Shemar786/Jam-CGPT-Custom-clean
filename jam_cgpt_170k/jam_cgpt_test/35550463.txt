		for (int i = 0; i < 8; i++) {

				port.append("0");
			} else {
				port.append("1");
			}
		}
		String mask1 = port.substring(0, 4);
		String mask2 = port.substring(4);
		String maskHex1 = Integer.toHexString(binaryStringToInt(mask1));
		String maskHex2 = Integer.toHexString(binaryStringToInt(mask2));
		return maskHex2 + maskHex1;
	}
