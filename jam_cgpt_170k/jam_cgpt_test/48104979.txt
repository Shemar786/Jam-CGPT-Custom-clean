TDAT:     private String toString(Partition partition) {
        String s = "";
        for (int i = 1; i <= partition.getNumBlocks(); i++) {
            s += i;
            s += ":";
            s += toString(partition.getBlock(i));
            if (i < partition.getNumBlocks())
                s += ",";
        }
        return s;
    }
