TDAT:     static protected String hostFromUriStr(String uriStr) {
        debugOut("hostFromUriStr(\"" + uriStr + "\")");
        String host = null;
        URI uri;
        try {
            uri = new URI(uriStr);
            host = uri.getHost();
        } catch (URISyntaxException e) {
            debugOut(e.getMessage());
        }
        return host;
    }
