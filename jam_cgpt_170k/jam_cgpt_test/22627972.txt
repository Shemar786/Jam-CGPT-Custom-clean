				adminData.put(Keys.REPOSITORY_IDENTIFIER, value);
			}
			else {
				adminData.update(Keys.REPOSITORY_IDENTIFIER, value);
			}
		} catch (Exception e) {
			prtlnErr("Error saving serialized REPOSITORY_IDENTIFIER: " + e);
		}
	}
