        String timeoutS = servletConfig.getInitParameter("timeout");
        int timeout = 120*1000; //2 min

            timeout = Integer.parseInt(timeoutS);
        }catch (NumberFormatException nfe) {
            //ignore, we have a default value
        }
        return timeout;
    }
