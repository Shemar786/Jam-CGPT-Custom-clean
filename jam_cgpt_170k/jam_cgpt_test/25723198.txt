TDAT: 	public void paintSmallTag(Graphics g, TGPanel tgPanel, int tagX, int tagY, Color backCol, Color textCol, char character) {
		g.setColor(backCol);
		g.fillRect(tagX, tagY, 8, 8);
		g.setColor(textCol);
		g.setFont(SMALL_TAG_FONT);
		g.drawString("" + character, tagX + 2, tagY + 7);
	}
