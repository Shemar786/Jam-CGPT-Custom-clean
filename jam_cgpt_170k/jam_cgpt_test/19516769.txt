		StringBuilder res = new StringBuilder();
		
		// table definition
		res.append("TABLE:" + elementType + "\n");
		
		// column headers
		for(String attr : attributes) {
			res.append("," + attr);
		}
		res.append("\n");
		
		// values
		for(Map<String,String> instance : instances) {
			for(String attr : attributes) {
				res.append(",");
				String value = instance.get(attr);
				
				if(value == null && attr.startsWith("*"))
					value = instance.get(attr.substring(1));
				
				if(value != null)
					res.append(value);
			}
			res.append("\n");
		}
		
		return res.toString();
	}
