				if (_myReader != null) {
					try {
						freq = 0;
						for (int i = 0; i < fields.size(); i++) {
							freq += _myReader.docFreq(new Term((String) fields.get(i), term));
						}
					} catch (Exception e) {}
				}
			}
		}
		return freq;
	}
