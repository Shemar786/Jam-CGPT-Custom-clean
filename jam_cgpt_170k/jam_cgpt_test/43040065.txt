TDAT:     public void setModel(T model) {
        // unregister old model
        if (this.model != null) {
            for (Listener l : properties)
                l.unsetModel(this.model);
        }
        this.model = model;
        // register new model
        if (this.model != null)
            for (Listener l : properties)
                l.setModel(this.model);
    }
