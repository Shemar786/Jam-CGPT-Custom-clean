TDAT:   static public StringBuffer toUpper(String str) {
    StringBuffer res = new StringBuffer(str);
    for (int i=0; i<str.length(); i++) {
      res.setCharAt(i,Character.toUpperCase(str.charAt(i)));
    }
    return res;
  }
