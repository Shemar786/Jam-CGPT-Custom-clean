TDAT:     public void setSubjectString(String inString) {
        StringTokenizer st = new StringTokenizer(inString,"\n");
        if (!st.hasMoreElements()) return;
        subjects = null;
        subjects = new ArrayList();
        
        while (st.hasMoreElements()) {
            Subject su = new Subject(st.nextToken());
            subjects.add(su);
        }
    }
