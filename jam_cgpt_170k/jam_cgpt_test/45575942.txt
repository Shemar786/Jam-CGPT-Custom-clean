        double[][] a;
        double[][] b;
        a = A.getArray();
        b = B.getArray();
        double dp = 0.0;
        for (int i = 0; i < a[0].length; i++) {
            dp = dp + (a[0][i] * b[i][0]);
        }
        return dp;
