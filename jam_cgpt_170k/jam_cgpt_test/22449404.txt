TDAT:     private String generateVariableName(LPVariable variable) {
        String result = variable.getName();
        if (result == null || result.length() == 0) {
            result = "x" + this.varCounter++;
        }
        else if (!Character.isLetter(result.charAt(0))) {
            result = "x" + result;
        }
        
        return result;
    }
