		int pos = 1;

		int value = this.entries.first().getValue();
		for (Entry<String, Integer> entry : this.entries) {
			if (value != entry.getValue()) {
				value = entry.getValue();
				pos++;
			}
			if (entry.getKey().equals(key)) {
				return pos;
			}
		}
		return -1;
	}
