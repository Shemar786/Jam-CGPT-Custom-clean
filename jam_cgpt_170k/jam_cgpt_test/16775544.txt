		int n = this.getCount();
		int nextindex = 0;

		double tempTimes[] = null;
		for (TimeSeries sery : this.series) {
			tempTimes = sery.getTimesRef();
			if(tempTimes!=null){
				seriesLength = tempTimes.length;
				System.arraycopy(sery.getTimesRef(), 0, times, nextindex, seriesLength);

			}
		}
		return new Vector(times);
	}
