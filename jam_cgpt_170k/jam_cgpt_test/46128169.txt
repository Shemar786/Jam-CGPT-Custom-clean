TDAT:    private Stack grabPortData(){
      Stack returnStack = new Stack();
      short porttype = this.portType(); //Get the port type
      if(porttype == PORT_USB){
         returnStack = this.grabUSBDefaultPortData();
      }
      else{
         returnStack = this.grabSerialPortData();
      }
      return returnStack;
   }
