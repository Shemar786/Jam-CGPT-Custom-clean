TDAT:     public List getSelectedValues() {
        List list = new ArrayList();
        Iterator iter = valueMap.keySet().iterator();
        while (iter.hasNext()) {
            JCheckBox checkbox = (JCheckBox)iter.next();
            if (checkbox.isSelected()) {
                String value = (String)valueMap.get(checkbox);
                list.add(value);
            }
        }
        return list;
    }
