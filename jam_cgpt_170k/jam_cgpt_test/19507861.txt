		assertDeltas(

		);

		clearDeltas();
		createRodinFile("/P/X.test");
		assertDeltas(
				"Unexpected delta after adding /P/p/X.test",
				"P[*]: {CHILDREN}\n" + 
				"	X.test[+]: {}"
		);
	}
