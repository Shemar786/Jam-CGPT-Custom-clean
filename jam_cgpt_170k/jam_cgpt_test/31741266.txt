TDAT: 	public void startHeader(Request request) {
		try {
			sendStatus(HTTP.OK);
			sendBasicHeaders();

			sendHeaderEntry("Last-Modified: "
					+ HTTP.formatDate(new Date(request.getRequestFile()
							.lastModified())));
			sendHeaderEntry("Content-Type: " + request.getMimeType());
			

		} catch (IOException e) {
			e.printStackTrace();
		}
	}
