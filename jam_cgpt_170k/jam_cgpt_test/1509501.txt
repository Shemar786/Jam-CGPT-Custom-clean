            return;
        }

        if (!hasFocus()) {
            setFocus(true);
            return;
        }

        currentPosition += granularity;

        if (currentPosition >= polysomnograph.getEpochDuration() &&
                canBeDisplayed()) {
            resetPosition();
            setFocus(false);
            polysomnograph.incrementStartRecord();
        }

        positionChanged();
    }
