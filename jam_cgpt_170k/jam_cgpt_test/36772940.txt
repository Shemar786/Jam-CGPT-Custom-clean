TDAT:     protected List arrayToList(Object[] array) {
        List<T> result = null;

        if (array != null && array.length > 0) {
            result = new ArrayList<T>();
        }

        for (int i = 0; i < array.length; ++i) {
            T tmp = (T) array[i];
            result.add(tmp);
        }

        return result;
