		if (node != null){
			NamedNodeMap attributes = node.getAttributes(); 
			for (int i = 0; i < attributes.getLength(); i++) { // Loop through Attributes
				Node a = attributes.item(i);
				if (ATTRNAMEVALUE.equals( a.getNodeName() )) {
					result = a.getNodeValue();
				}
			}
		}
		return result;
	}
