TDAT:   public int getIntValue() {
    switch (type) {
      case ModulePropertyType.TYPE_BOOLEAN:
        return ((Boolean) value).booleanValue() ? 1 : 0;
      case ModulePropertyType.TYPE_STRING:
        try {
          return new Double((String) value).intValue();
        }
        catch (Exception e) {
          return 0;
        }
      default:
        return ((Number) value).intValue();
    }
  }
