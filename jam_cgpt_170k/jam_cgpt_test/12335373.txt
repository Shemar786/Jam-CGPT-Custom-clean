		Set<ServerID> stack = new HashSet<ServerID>();
		for (ServerID node : reversePath) {
			if (stack.contains(node)) {
				while (!realPath.getLast().equals(node)) {
					ServerID previous = realPath.removeLast();
					stack.remove(previous);
				}

				stack.add(node);
				realPath.addLast(node);
			}
		}
		this.reversePath = realPath;
