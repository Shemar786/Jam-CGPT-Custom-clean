TDAT: 	private void addSelectedGroup(String ou){
		//to prevent the place holder item from provoking a response in the interface
		if(!ou.equals("0")){
			LdapGroup movedGroup = (LdapGroup)availableGroups.remove(ou);
			if(selectedGroups.containsKey("0")){
				selectedGroups.remove("0");
			}
			selectedGroups.put(ou,movedGroup);
			if(availableGroups.isEmpty()){
				availableGroups.put(controller.getEmptyAvailableGroup().getOU(),controller.getEmptyAvailableGroup());
			}
		}
	}
