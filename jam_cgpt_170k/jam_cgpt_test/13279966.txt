		JMenu subMenu = null;
		// check if this menu already exists
		if (menuItemMap.containsKey(menuname)) {

			subMenu = (JMenu) menuItemMap.get(menuname);

		} else {
			// make a new menu
			subMenu = new JMenu(menuname);
			menuItemMap.put(menuname, subMenu);

		}

		subMenu.add(jMenuItem);

	}
