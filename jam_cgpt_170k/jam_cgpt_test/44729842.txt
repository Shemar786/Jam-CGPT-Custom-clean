			if (file.exists()) {
				try {
					comment = FilesContentReader.getBodyContent(file);
				} catch (Exception e) {
					log.error("Error loading comment", e);
					comment = "Error loading comment: " + e;
				}
			} else {
				log.warn("commentFile not found: " + file.getAbsolutePath());
			}
		}
