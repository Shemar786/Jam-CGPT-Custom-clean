		int rowInx = 0;

		for (; rowInx < tblInfo.size(); rowInx++) {
			Vector<Object> row = tblInfo.get(rowInx);
			if (row.get(0).equals(vid)) {

			}
		}
		if (rowInx >= tblInfo.size()) {
			rowInx = -1;
		}

		return rowInx;
	}
