TDAT:     public void groupCurrentSelection() {
        if (!selection.isEmpty()) {

            Group gr = new Group(new View(Window.sceneGraph.getView()));

            int i = selection.size();
            while (i-- > 0) {
                SceneGraph sg = selection.get(i);
                gr.add(sg);
            }

            Window.sceneGraph.add(gr);

            selection.clear();
            selection.add(gr);

            repaintPanel();
        }
    }
