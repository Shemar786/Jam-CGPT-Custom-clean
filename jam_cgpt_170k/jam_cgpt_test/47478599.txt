		if (removed.length > 0)
			type |= IModelProviderEvent.MODELS_REMOVED;
		IModel[] added = state.getTargetModels();
		if (added.length > 0)
			type |= IModelProviderEvent.MODELS_ADDED;
		modelsChanged(new ModelProviderEvent(state, type, added, removed, new IModel[0]));

	}
