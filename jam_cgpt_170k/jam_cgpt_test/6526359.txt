TDAT:    public boolean belongsToCategory(Category aCategory) {
       boolean result = false;
       if (this.categories.contains(aCategory)) {
           result =  true;
       } else {
           Iterator it = aCategory.getChildren().iterator();
           while (it.hasNext()) {
               if (this.belongsToCategory((Category) it.next())) {
                   result = true;
               }
           }
       }
       return result;
   }
