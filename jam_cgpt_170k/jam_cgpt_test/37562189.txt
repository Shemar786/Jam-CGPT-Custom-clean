TDAT: public String getItem (int index) {
	checkWidget ();
  int count = getItemCount();
  if (!(0 <= index && index < count)) error (SWT.ERROR_CANNOT_GET_ITEM);
  try {
    return (String)((CCombo)handle).getItemAt(index);
  } catch(Exception e) {
    error (SWT.ERROR_INVALID_RANGE);
  }
  return "";
}
