		int count = 0;
		int place = children.size();
		for (int i=0; i<children.size(); i++) {
			String name = ((XMLNode)children.get(i)).getName();
			if (name.equals(node.getName())) {
				count++;
				place = i+1;
			}	
		}
		node.occurs = count;
		return place;
	}
