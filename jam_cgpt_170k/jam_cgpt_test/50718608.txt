TDAT:    private void doExport() {

      if (!pagesValid())
         return;

      workingThread = null;

      if (cvtType.getSelectedIndex() == 0)
         workingThread = new Thread(new Runnable () {
            public void run() {
               cvtToPDF();
            }
         });
      else
         workingThread = new Thread(new Runnable () {
            public void run() {
               cvtToText();
            }
         });

      workingThread.start();
   }
