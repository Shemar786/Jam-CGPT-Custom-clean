TDAT: 	private void copy(BufferedReader reader, BufferedWriter writer) throws IOException {
        String line = null;
        while ((line=reader.readLine()) != null) {
        	// if line is not a comment, copy line to writer
        	if (!(line.startsWith("<?"))){
        		writer.write(line);
        		writer.newLine(); 
        	}
        }

	}
