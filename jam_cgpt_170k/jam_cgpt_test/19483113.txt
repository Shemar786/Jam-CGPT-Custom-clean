		if (problem == null)
			throw new IllegalArgumentException(ExceptionMessages.NULL_ARGUMENT);

		String message = problem.getMessage();
		String category = problem.getCategory();

		switch (problem.getSeverity()) {
		case ERROR:
			MessageDialog.openError(shell, category, message);
			break;
		case WARNING:
			MessageDialog.openWarning(shell, category, message);
			break;
		case INFORMATION:
			MessageDialog.openInformation(shell, category, message);
			break;
		}
