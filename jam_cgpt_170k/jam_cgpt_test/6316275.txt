TDAT: 	public void selectItem(NameAndPackage namePackage)
	{		
		try
		{
			TreeItem item = findTreeItem(namePackage);
			if(item != null)
			{
				tree.setSelection(new TreeItem[] { item });
				tree.showSelection();
				notifySelected((IRItem) item.getData());
			}
		}
		catch (TreeException e)
		{
			MessageBoxAndLog.show(tree.getShell(), "Project", "Selecting the item "+namePackage+" in the tree got", e);
        }        
	}
