TDAT:     public int getResultSetCount(ResultSet rs) {
        int count = 0;
        try {
            rs.beforeFirst();
            while (rs.next()) {
                count++;
            }        
            rs.beforeFirst();
        } catch (Exception e) {
            log.error(Constants.EXCEPTION_CAUGHT, e);            
        }
        return count;
    }
