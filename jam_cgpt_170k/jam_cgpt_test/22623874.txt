TDAT: 	public List getDirs() {
		if (recordsDir == null) {
			prtln("getDirs: recordsDir is null");
			return null;
		}
		File[] files = recordsDir.listFiles();
		ArrayList dirs = new ArrayList();
		for (int i = 0; i < files.length; i++) {
			File file = files[i];
			if (file.isDirectory()) {
				dirs.add(file.getName());
			}
		}

		return dirs;
	}
