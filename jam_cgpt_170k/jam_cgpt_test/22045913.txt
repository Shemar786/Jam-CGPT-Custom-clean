	int pos = textPane.getCaret().getDot();
	java.util.ListIterator<Integer> i;
	for( i = navigator.dots().listIterator(); i.hasNext(); ){
	    int p = i.next();
	    if( p >= pos ){
		i.previous();
		break;
	    }
	}
	dotPosition = i.hasPrevious() ? i.previous() : 0;
    }
