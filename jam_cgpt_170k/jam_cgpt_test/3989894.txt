TDAT:         private String literal(Data list) {
            StringBuffer  buffer = new StringBuffer();

            buffer.append("(");
            for (int i = 0; i < list.arraySize(); i++) {
                if (i > 0) {
                    buffer.append(",");
                }
                buffer.append(literal(list.get(i)));
            }
            buffer.append(")");
            return buffer.toString();
        }
