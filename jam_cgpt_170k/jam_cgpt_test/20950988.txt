		Calendar cal = Calendar.getInstance();

		cal.setTime(date);

		buffer[pos] = (byte)(cal.get(Calendar.YEAR) - 1900);
		buffer[pos + 1] = (byte)(cal.get(Calendar.MONTH) + 1);
		buffer[pos + 2] = (byte)(cal.get(Calendar.DAY_OF_MONTH));
	}
