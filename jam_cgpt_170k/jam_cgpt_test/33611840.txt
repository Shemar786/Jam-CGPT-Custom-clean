TDAT:     public int flush() throws InterruptedException{
        int count = 0;
        while(this.getReportQueue().size()!=0){
            Report report = this.getReportQueue().getFirst();
            if(this.flush(report)){
                this.getReportQueue().remove(report);
                count ++;
            }
        }
        return count;
    }
