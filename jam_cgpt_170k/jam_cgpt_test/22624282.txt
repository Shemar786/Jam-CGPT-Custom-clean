		if (securityRole.equals("trustedIp")) {
			String[] trustedIps = repositoryManager.getTrustedWsIpsArray();
			if (trustedIps == null)
				return false;

			String IP = request.getRemoteAddr();
			for (int i = 0; i < trustedIps.length; i++)
				if (IP.matches(trustedIps[i]))
					return true;
		}
		return false;
	}
