TDAT: 	private Time simulate() throws SimulationException {
		Time current = startSimulation();
		try {
			if (endTime.isLT(current)) {
				throw new SimulationException("Requested time " + endTime
						+ " is smaller than current time " + current + "!");
			}
			while (isRunning()) {
				if (current.isGE(endTime)) {
					break; // The requested end time has been reached
				}
				current = continueSimulation(current, endTime);
				Thread.yield();
			}
		} finally {
			finishSimulation();
		}
		return currentTime();
	}
