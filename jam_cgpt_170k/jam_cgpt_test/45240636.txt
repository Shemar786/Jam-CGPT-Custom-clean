		for(Entry<Long, Integer> friend : player.getFriendList()) {
			int world = friend.getValue();
			s.addLong(friend.getKey());
			s.addByte((byte)(world == Config.SERVER_NUM ? 99 : world));
		}
		packets.add(s.toPacket());
	}
