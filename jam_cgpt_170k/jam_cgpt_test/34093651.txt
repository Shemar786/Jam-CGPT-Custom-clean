TDAT: 	private void writeFilters(Properties props, String keyPrefix, Collection<String> filters) {
		int counter = 0;
		for (String s : filters) {
			props.setProperty(keyPrefix + counter, s);
			counter++;
		}
		// remove obsolete keys from the properties file
		boolean keyFound = true;
		while (keyFound) {
			String key = keyPrefix + counter;
			String property = props.getProperty(key);
			if (property == null) {
				keyFound = false;
			} else {
				props.remove(key);
			}
		}
	}
