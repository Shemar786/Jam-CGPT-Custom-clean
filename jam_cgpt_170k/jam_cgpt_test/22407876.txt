        if ((errorMessage == null) && (fileExtension != null)) {
            int startLineNumber =
                    str.indexOf(fileExtension) + fileExtension.length() + 1;

            int endLineNumber = str.indexOf(": ", startLineNumber);

            if (endLineNumber > 0) {
                errorMessage = str.substring(endLineNumber + 2);
            }
        }

        return errorMessage;
    }
