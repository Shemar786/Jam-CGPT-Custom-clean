TDAT:     private String implode(String[] array, String separator) {
        String str = "";
        int i = 0;
        while(i < array.length - 1) {
            str += array[i] + separator;
            i++;
        }
        if(i < array.length) {
            str += array[i];
        }
        return str;
    }
