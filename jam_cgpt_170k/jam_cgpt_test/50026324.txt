    int a = getAlignment(type);
    int b = (int) ((bytecounter - preallocated) % a);
    if (0 == b) { return; }
    a = (a - b);
    if (preallocated > 0) {
      paofs += a;
      preallocated -= a;
    } else {
      appendBytes(padding[a]);
    }
  }
