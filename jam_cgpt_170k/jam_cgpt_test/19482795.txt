			res2.save(Collections.EMPTY_MAP);
			res2.getContents().add(element);
			res2.save(Collections.EMPTY_MAP);
			
			// synchronise modification stamp (used to prevent synchronisation loop)
			file2.revertModificationStamp(file1.getModificationStamp());
		}
