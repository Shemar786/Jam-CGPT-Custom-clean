TDAT:     public Object convert(Context context, Object file) throws ConversionException {
        try {
            return new FileInputStream((File) file);
        } catch (FileNotFoundException e) {
            throw new ConversionException("Cannot find file '" + ((File) file).getName() + "'", e);
        }
    }
