TDAT: public void setMaximized (boolean maximized) {
	checkWidget ();
  CShell cShell = (CShell)handle;
  if(maximized) {
    cShell.setExtendedState(CShell.MAXIMIZED_BOTH);
  } else {
    cShell.setExtendedState(cShell.getExtendedState() & ~CShell.MAXIMIZED_BOTH);
  }
}
