TDAT: 	private int calculateDotLevel(String toCalc) {
		int level = 0;
		char ch;
        for ( int i = 0;  i < toCalc.length();  i++ ) {
            ch = toCalc.charAt(i);  // Get the i-th character in str.
            if ( ch == '.'){
               level++;
            }
        }     
		
		return level;
	}
