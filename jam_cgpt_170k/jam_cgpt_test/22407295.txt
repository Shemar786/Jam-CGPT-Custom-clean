TDAT:     protected Node findFirstChildNode(final Node parent, final String name) {
        final NodeList list = parent.getChildNodes();
        for (int i = 0; i < list.getLength(); i++) {
            final Node currentNode = list.item(i);
            if (name.equalsIgnoreCase(currentNode.getLocalName())) {
                return currentNode;
            }
        }

        return null;
    }
