			if (act.getIdentifier().equals(action.getIdentifier())) {
				System.out.println("Action with identifier: " + act.getIdentifier() + " already added.");
				return;
			}
		}

		System.out.println("ServiceContextManager ADDING Action (" + action.getClass().getName() + ")");

		actions.add(action);

		fireActionsChanged(action, ActionsChangedEvent.Type.ADDED);
	}
