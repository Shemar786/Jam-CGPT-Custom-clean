TDAT: 	public void actionPerformed(ActionEvent e) {
		if (EDIT.equals(e.getActionCommand())) {
			//The user has clicked the cell, so
			//bring up the dialog.
			wpListDlg.setVisible(true);

			//Make the renderer reappear.
			fireEditingStopped();
		}
		else if (wpListDlg.isOkEvent(e)) {
			stopCellEditing();
		}
		else if (wpListDlg.isCancelEvent(e)) {
			cancelCellEditing();
		}
	}
