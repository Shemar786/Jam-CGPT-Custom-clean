TDAT:     private String byteArrayToString(byte[] array) {
        StringBuffer res = new StringBuffer();
        for (int i=0;i<array.length;++i) {
            String hexString = Integer.toHexString(array[i] & 0xFF);
            res.append(hexString + " ");
        }
        return res.toString();
    }
