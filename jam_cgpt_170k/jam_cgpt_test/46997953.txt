TDAT:     private String skipSpaces(String str) {
        int i = 0;
        int len = str.length();
        StringBuffer nstr = new StringBuffer(len);

        while (i < len) {
            if (str.charAt(i) != ' ') {
                nstr.append(str.charAt(i));
            }
            i++;
        }

        return nstr.toString();
    }
