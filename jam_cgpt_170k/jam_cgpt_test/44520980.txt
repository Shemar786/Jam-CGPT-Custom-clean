			if (name.equals(t.name())) {
				int snum=t.getStat("Number");
				if (snum>=num) {
					c+=num;
					t.remove(num);
					break;
				}
                t.remove();
                num-=snum;
                c+=snum;
                i--; // step back one since stack is removed
			}
		}
