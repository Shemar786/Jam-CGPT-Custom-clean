				.toURI(), file.lastModified(), file.length()));
		if (file.isDirectory()) {
			File[] files = file.listFiles();
			for (int i = 0; i < files.length; i++) {
				collectFileLinksRecursive(files[i], fileLinks);
			}
		}
	}
