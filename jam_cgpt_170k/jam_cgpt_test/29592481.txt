TDAT:   static public String removeWhitespace( String s) {
    int len = s.length();
    StringBuffer b = new StringBuffer( len);
    for (int i=0; i< len; i++) {
      char c = s.charAt(i);
      if (!Character.isWhitespace(c))
        b.append(c);
    }
    return b.toString();
  }
