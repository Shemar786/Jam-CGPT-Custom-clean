        long now = System.currentTimeMillis();
        long start = mAuth.getEffectiveDate().getTime();
        long end = mAuth.getExpirationDate().getTime();
                
        if (start<=now && end >= now) { 
            return true;
        }
        else {
            return false;
        }
    }
