TDAT:     public void removePossibleDefender(String name, boolean penalize) {
        String currName = name.toLowerCase();
        OpsChickenThread currT = chickenThreads.get(currName);
        if (currT == null) {
            return;
        }

        if (penalize) {
            currT.doPenalty();
        }
        currT.stopChicken();
        chickenThreads.remove(currName);
    }
