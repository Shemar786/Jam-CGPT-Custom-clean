		box.addItem("None");

		if (ans.size() > 0) {
			AttributeName[] array = ans.toArray(new AttributeName[0]);
			Arrays.sort(array);
			for (AttributeName an : array) {
				box.addItem(an);
			}
			box.revalidate();
		}
	}
