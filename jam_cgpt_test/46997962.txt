TDAT:     private int match(String exp, int index) {
        int len = exp.length();
        int i = index;
        int count = 0;

        while (i < len) {
            if (exp.charAt(i) == '(') {
                count++;
            } else if (exp.charAt(i) == ')') {
                count--;
            }

            if (count == 0) {
                return i;
            }

            i++;
        }

        return index;
    }
