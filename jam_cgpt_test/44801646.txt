TDAT:     public Mib load(String name) throws IOException, MibLoaderException {
        MibSource  src;
        Mib        mib;

        mib = getMib(name);
        if (mib == null) {
            src = locate(name);
            if (src == null) {
                throw new FileNotFoundException("couldn't locate MIB: '" +
                                                name + "'");
            }
            mib = load(src);
        }
        return mib;
    }
