		if (c.moveToFirst()) {
			ret = c.getInt(0);
		} else {
			db.insert(GEORSS_TABLE, null, getContentValues(feed));
			c.requery();
			c.moveToFirst();
			ret = c.getInt(0);
		}
		c.close();
		return ret;
