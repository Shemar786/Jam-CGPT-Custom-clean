		issue.setPriority((Priority) priority.getSelectedItem());
		issue.setSummary(summary.getText());
		issue.setDescription(description.getText() + "\n\nEmail: "
				+ email.getText());

		// submit issue
		service.createIssue(issue);

		// close dialog
		try {
			this.setClosed(true);
		} catch (PropertyVetoException e1) {
			ErrorDialog.show(e1);
		}
	}
