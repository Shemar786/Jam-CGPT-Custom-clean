TDAT: 	private void processDocumentDelta(IRodinElementDelta delta) {

		int kind = delta.getKind();
		if (kind == ADDED || kind == REMOVED) {
			String[] items = getDocumentComboLabels();
			final ComboBoxCellEditor cellEditor = page.getComboBoxCellEditor();
			if (cellEditor != null) {
				cellEditor.setItems(items);
			}
		}

		if (kind == REMOVED) {
			refreshAll = true;
		}
	}
