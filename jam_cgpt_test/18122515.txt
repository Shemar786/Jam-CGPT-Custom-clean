TDAT: 	private String getPreference(String name, IProject project) {
		String toReturn = null;
		if(project != null) {
			try {
				 toReturn = project.getPersistentProperty(new QualifiedName(PLUGIN_ID, name));
			} catch (CoreException e) {
				// Ignore as this means we don't have any project properties...
			}
		}
		if(toReturn == null) {
			toReturn = this.getPreferenceStore().getString(name);
		}
		return toReturn;
	}
