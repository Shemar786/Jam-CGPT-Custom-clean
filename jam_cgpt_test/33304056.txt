TDAT:     public String getAsList(String key, String separator) {
        StringBuffer sb = new StringBuffer();
        Iterator it = (Iterator)get(key);
        while (it != null && it.hasNext()) {
            if (sb.length() != 0) 
                sb.append(separator);
            sb.append(it.next().toString());
        }
        return sb.toString();
    }
