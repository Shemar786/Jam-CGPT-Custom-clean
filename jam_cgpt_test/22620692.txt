TDAT: 	protected void insertHtmlComment(Element e, String s) {
		prtln("\ninsertHtmlComment()");
		Element comment = DocumentHelper.createElement("st__htmlComment");
		comment.setText(s);
		Element parent = e.getParent();

		if (parent != null) {
			List children = parent.elements();
			int index = children.indexOf(e);
			children.add(index, comment);
		}
		else {
			// prtlnErr("PARENT NOT FOUND");
		}
	}
