		if (c.size() == 0) {
			min = 0.0;
			max = 1.0;
		} else {
			Iterator<? extends Number> it = c.iterator();
			double n = it.next().doubleValue();
			min = n;
			max = n;
			while (it.hasNext()) {
				n = it.next().doubleValue();
				if (n < min)
					min = n;
				else if (n > max)
					max = n;
			}
		}
	}
