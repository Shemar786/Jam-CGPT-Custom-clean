		try {
			// get rid of the old font
			Font font = getNameLabel().getFont();
			font.dispose();
			
			boolean italic = model().isAbstract() || model().isInterface();
			
			// create the new font
			getNameLabel().setFont(PlugIn.getFontPreference(P_FONT, italic));
		} catch (JavaModelException e) {
			e.printStackTrace();
		}
	}
