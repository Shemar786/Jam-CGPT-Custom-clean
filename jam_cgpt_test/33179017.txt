	StringBuffer printable = new StringBuffer(notPrintable.length());
	for (int i = 0; i < notPrintable.length(); ++i) {
	    char c = notPrintable.charAt(i);
	    switch (c) {
		case '\n': printable.append("\\n"); break;
		case '\t': printable.append("\\t"); break;
		default: printable.append(c); break;
	    }
	}

	return printable.toString();
