    if (target == this) {
      return true;
    } else {
      // depth first search the adjacency list
      setVisited(true);

      for (Enumeration e = adjacency.keys(); e.hasMoreElements(); ) {

	if (!adj.getVisited()) {

	  }
	}
      }

    }
