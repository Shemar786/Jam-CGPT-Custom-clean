		{
			list = tabs.nextElement().getTabpanels().getChildren();
			Iterator<?> itr = list.iterator();
			while(itr.hasNext()){
				tempPanel = (Tabpanel)itr.next();
				tempView = (View)tempPanel.getChildren().get(0);
				tempView.onCreate();
			}
		}
	}
