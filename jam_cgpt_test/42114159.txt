TDAT: 	public void hit(Person player, int damage) {
		if(isDead()) return ;
		if(isGodMode()) return ;
		
		if(getArmor() > 0) {
			armor -= damage;
			if(armor < 0) armor = 0;
		}else{
			health -= damage;
		}
		
		if(player != null && health <= 0) {
			player.killedOpponent(this.getPlayerName());
			died(player.getPlayerName(), player.getCameraPosition());
		}
	}
