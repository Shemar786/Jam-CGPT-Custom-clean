TDAT: 	public String evalCommand(String cmdName, String[] words) {
		if (ScriptCommands.isBuiltinCommand(cmdName))
			return ScriptCommands.evalBuiltinCommand(cmdName, words, mDepth);

		if (getFunctionsStore().contains(cmdName)) {
			ScriptFunction func = (ScriptFunction) getFunctionsStore()
					.getVariable(cmdName, mDepth);
			return func.evalFunction(words, mDepth);
		}
		return "Unknown command " + cmdName;
	}
