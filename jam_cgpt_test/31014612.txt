TDAT:   static public StringBuffer toLower(String str) {
    StringBuffer res = new StringBuffer(str);
    for (int i=0; i<str.length(); i++) {
      res.setCharAt(i,Character.toLowerCase(str.charAt(i)));
    }
    return res;
  }
