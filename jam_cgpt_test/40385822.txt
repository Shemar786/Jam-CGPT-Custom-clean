		if (texture == null) {
			texture = this.getTexture(image, GL.GL_TEXTURE_2D, // target
			        GL.GL_RGBA, // dst pixel format
			        GL.GL_LINEAR, // min filter (unused)
			        GL.GL_LINEAR);
			
			this.table.put(image, texture);
		}
		
		return texture;
	}
