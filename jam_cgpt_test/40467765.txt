		if(useImages){
			switch(imgAlign){
			case GAlign.LEFT:
				imageAlign = imgAlign;
				imgAlignX = PADH;
				break;
			case GAlign.RIGHT:
				imageAlign = imgAlign;
				imgAlignX = width - btnImgWidth - PADH;
				break;
			case GAlign.CENTER:
				if(text.length() == 0){
					imageAlign = imgAlign;
					imgAlignX = (width - btnImgWidth)/2;
				}
				else {
					imageAlign = GAlign.LEFT;
					imgAlignX = PADH;
				}
				break;
			}
		}
		calcAlignX();
	}
