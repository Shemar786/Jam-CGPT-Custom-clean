TDAT:     public void addRule(String path, IXMLAction action) {
        if (!paths.contains(path)) {
            paths.add(path);
        }
        List list = null;
        if (!actions.containsKey(path)) {
            list = new ArrayList();
            actions.put(path, list);
        } else {
            list = (List)actions.get(path);
        }
        list.add(action);
    }
