TDAT: 	private URL makeFileURL(String first, String... parts) {
		if (first.startsWith("http")) {
			for (String p : parts) {
				first += "/" + p;
			}
			try {
				return new URL(first);
			} catch (MalformedURLException e) {
				throw new Error(e);
			}
		} else {
			File f = new File(first);
			for (String p : parts) {
				f = new File(f, p);
			}
			try {
				return f.toURI().toURL();
			} catch (MalformedURLException e) {
				throw new Error(e);
			}
		}
	}
