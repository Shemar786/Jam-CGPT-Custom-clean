		InputStream ins = null;
		// check explicit context-param['logging-config']
		String param = servletContext.getInitParameter(LOGGING_CONTEXT_PARAM);
		if (param != null) {
			ins = servletContext.getResourceAsStream(param);
		}
		// check /WEB-INF/logging.properties
		if (ins == null) {
			ins = servletContext.getResourceAsStream("/WEB-INF/logging.properties");
		}
		// check classpath: logging.properties
		if (ins == null) {
			ClassLoader cl = Thread.currentThread().getContextClassLoader();
			ins = cl.getResourceAsStream("logging.properties");
		}
		return ins;
	}
