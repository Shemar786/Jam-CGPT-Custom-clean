		MyCallback callback = new MyCallback();

		try {
			/**
			 * parses the html/xml file, as pointed by the Reader, and handles
			 * it as specified by the callback. The ignoreCharSet equals false.
			 */
			new ParserDelegator().parse(r, callback, false);
		} catch (IOException ex) {
			logger.error("error parsing file", ex);
			return "";
		}

		return callback.htmlString;
	}
