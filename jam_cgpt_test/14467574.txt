TDAT:   public TableCellEditor getCellEditor(int row, int col) {
    Object value = getValueAt( row, col );

    if( value instanceof BelEnumerationAttribute ) {
      this.enumEditor.setData( (BelEnumerationAttribute) value );
      return new DefaultCellEditor( this.enumEditor );
    }

    return super.getCellEditor(row, col);
  }
