TDAT:     public void activate(Planet p) {

        if (p == null) {
            return;
        }

        if (mp.getPPanel() != null && mp.getPPanel().getPlanet() != p) {

            mp.getPPanel().update(p);
            conf.planetID = p.getId();
            mp.repaint();

            saveMapSelection(p);
        }
    }
