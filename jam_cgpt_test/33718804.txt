TDAT:     public String GetAutismGroupName(int gid){
        if (this.groupList==null){
            return null;
        }else{
            Iterator itr = this.groupList.iterator();
            while(itr.hasNext()){
                AutismGroup ag = (AutismGroup)itr.next();
                if(ag.GetGroupId()==gid){
                    return ag.GetGroupName();
                }
            }   
        }        
        return null;
    }
