      int w = calendar.get(Calendar.DAY_OF_WEEK);
      int d = calendar.get(Calendar.DAY_OF_MONTH);
      while (true) {
        if (d <= 1 && w == calendar.getFirstDayOfWeek()) break;
        d--; w--;
        w = (w < 1) ? 7 : w;
      }
      return d;
    }
