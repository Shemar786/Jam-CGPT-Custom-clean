TDAT:     protected int checkString(int initialState) {
        if (ch != '\"') {
            return -1;
        }

        readNextChar();

        while (ch != '\"' && ch != '\n' && ch != EOF) {
            readNextChar();
        }

        if (ch != EOF) {
            readNextChar();
        }

        return BasicStyleMap.STRING;
    }
