TDAT: public void setBackgroundMode (int mode) {
  checkWidget ();
  if(backgroundMode == mode) return;
  backgroundMode = mode;
  Control [] children = _getChildren ();
  for (int i = 0; i < children.length; i++) {
    children [i].updateBackgroundMode ();   
  }
  handle.repaint();
}
