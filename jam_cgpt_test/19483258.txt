TDAT: 	public void elementChanged(ElementChangedEvent event) {

		refreshAll = false;
		toRefresh = new LinkedList<IRodinElement>();

		processDelta(event.getDelta());

		if (page.getViewer() == null
				|| page.getViewer().getControl().isDisposed())
			return;

		Display.getDefault().syncExec(new Runnable() {
			public void run() {

				if (refreshAll)
					page.getViewer().refresh();
				else
					refreshTree(toRefresh);

				updateText();
			}
		});

	}
