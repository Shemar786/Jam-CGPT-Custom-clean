TDAT:     public void tableChanged(TableModelEvent e) {
        if (e.getType() == TableModelEvent.UPDATE) {// setValueAt will fire this event, do nothing to keep current sort
            fireTableChanged(e);
            return;
        }
        if (e.getType() == TableModelEvent.DELETE)
            updateDelete(e); // for delete keep current sort
        else {
            reallocateIndexes();
            fireTableChanged(e);
        }
    }
