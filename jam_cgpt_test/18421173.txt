TDAT: 	public Contact add(Date thisAppointment, Date nextAppointment, String note) throws ContactDateException  {
		Contact contact = new Contact();
		contact.setDate(thisAppointment);		
	    contact.setNextAppointment(nextAppointment);		
		contact.setNotice(note);
		contact.setCreatedBy((Dietician)m_UserHandler.getUser());
				
		DatabaseHandler dbH = DatabaseHandler.getInstance();
		m_UserHandler.getCurrentPatient().addContact(contact);
		dbH.savePatient(m_UserHandler.getCurrentPatient());
		
		return contact;
	}
