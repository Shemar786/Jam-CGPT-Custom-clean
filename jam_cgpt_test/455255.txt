		SparseDoubleMatrix2D mat1 =getMatrix(slice1);
		SparseDoubleMatrix2D mat2 =getMatrix(slice2);
		
		for (int i = 0; i < mat1.rows(); i++) {
			for (int j = 0; j < mat1.columns(); j++) {

			}
		}
