TDAT:   private boolean messageRegExpMatches(String messageRegExp, Throwable ite) {
    if (".*".equals(messageRegExp)) {
      return true;
    } else {
      if (ite.getMessage() == null) return false;
      else return Pattern.matches(messageRegExp, ite.getMessage());
    }
  }
