TDAT: 	private String getTitle(Model m, Resource r) {
		StmtIterator iter =  m.listStatements( new SimpleSelector( r, ATOM.title, (Literal) null ) );
		Literal title = null;
		while (iter.hasNext()) {
			Statement s = iter.nextStatement();
			title = s.getLiteral();
		}
		if (title == null) {
			return r.getURI();
		} else {
			return title.getString();
		}
	}
