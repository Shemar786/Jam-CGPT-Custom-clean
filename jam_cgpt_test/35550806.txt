TDAT: 	public void setPort(int data) {
		initPort();

		if ((data & 1) == 1)
			chkVlan.setSelected(true);
		if ((data & 2) == 2)
			chkMac.setSelected(true);
		if ((data & (2 * 2)) == 2 * 2)
			chkQos.setSelected(true);
		if ((data & (2 * 2 * 2)) == 2 * 2 * 2)
			chkFilter.setSelected(true);

		return;
	}
