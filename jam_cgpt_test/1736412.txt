TDAT: 	public void fillOutReasonCode() {
		List list = notificationReasonCodeService.findAllReasonCode();
		if (list == null || list.size() == 0) {
			reasonCodeList = null;
			return;
		}

		reasonCodeList = new ArrayList();
		for (int i = 0; i < list.size(); i++) {
			TblReasonCode item = (TblReasonCode) list.get(i);
			reasonCodeList.add(new SelectItem(item.getReasoncodeid(), item.getDescription()));
		}
	}
