        String[] properties = new String[environment.size()];

        Iterator iterator = environment.keySet().iterator();
        int i = 0;

        while (iterator.hasNext()) {
            String key = (String) iterator.next();

            Property property =
                    new Property(key, (String) environment.get(key));

            properties[i] = property.toString();
            i++;
        }

        localEnvProperties.setValues(this, properties);
    }
