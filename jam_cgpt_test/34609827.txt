        if (o == this) {
            return true;
        } else if (o == null) {
            return false;
        } else if (o instanceof MutableString) {
            return equals((MutableString) o);
        } else if (o instanceof String) {
            return equals((String) o);
        } else if (o instanceof CharSequence) {
            return equals((CharSequence) o);
        } else {
            return false;
        }
    }
