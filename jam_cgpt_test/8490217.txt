TDAT:   public Counter asCounter() {
    PriorityQueue<E> pq = clone();
    Counter<E> counter = new Counter<E>();
    while (pq.hasNext()) {
      double priority = pq.getPriority();
      E element = pq.next();
      counter.incrementCount(element, priority);
    }
    return counter;
  }
