TDAT: 	private String formatWithWildcard(String value, String prefix, String suffix) {
		if ((prefix != null) && (suffix != null)) {
			return prefix + value + suffix;
		} else if ((prefix != null)) {
			return prefix + value;
		} else if ((suffix != null)) {
			return value + suffix;
		} else {
			return value;
		}
	}
