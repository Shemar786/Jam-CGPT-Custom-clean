TDAT:     public void addPROCParam(PROCParam ce) {
        ce.setParent(this);
        //Set an inital location for this
        ce.setX(this.getX());
        ce.setY(this.getY());

        if(ce instanceof VAR){
            ((VAR) ce).setIsPROCParam(true);
        }

        params.add(ce);
    }
