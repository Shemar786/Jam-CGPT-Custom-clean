TDAT:     private int getColumnCount(UIComponent component) {
        int count;
        Object value = component.getAttributes().get("columns");
        if ((value != null) && (value instanceof Integer)) {
            count = (Integer) value;
        } else {
            count = 2;
        }
        if (count < 1) {
            count = 1;
        }
        return (count);
    }
