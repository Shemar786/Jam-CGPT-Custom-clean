TDAT:   private String getDirConstant(ServletConfig config, String initParam, String defaultValue) {
    String dir = config.getInitParameter(initParam);
    if ((dir == null) || (dir.trim().length() < 1)) {
      dir = defaultValue;
    } else {
      dir = dir.trim();
    }
    if (!dir.endsWith("/")) {
      dir += "/";
    }
    return dir;
  }
