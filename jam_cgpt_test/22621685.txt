		Pattern p = Pattern.compile(uidPat);
		Matcher m = p.matcher(dn);
		if (m.find()) {
			// prtln (" ... found it");
			return m.group(1);
		}
		else {
			// prtln (" ... didn't find it");
			return null;
		}
	}
