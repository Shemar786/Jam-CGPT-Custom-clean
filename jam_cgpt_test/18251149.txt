			char c = fullName.charAt(0);
			if (SqlUtil.LETTERS_AND_DIGITS.indexOf(c) < 0) {
				// quoting
				int end = fullName.substring(1).indexOf(c);
				if (end >= 0) {
					end += 1;
					int i = fullName.substring(end).indexOf('.');
					if (i >= 0) {
						return i + end;
					}
					return -1;
				}
			}
		}
		return fullName.indexOf('.');
	}
