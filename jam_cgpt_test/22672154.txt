TDAT: 	private void refreshCombo() {
		try {
			accounts = DAOFactory.getCurrentFactory().getAccountDAO().findAll();

			if (allowNull) {
				accounts.add(0, null);
			}
			
			accountCombo.removeAll();
			for (Account account : accounts) {
				if (account == null) {
					accountCombo.add("*** EMPTY ***");
				} else {
					accountCombo.add(
						MessageFormat.format(
							"{0}",
							account.toString()
						)
					);
				}
			}
		} catch (DAOException e) {
			accounts = null;
			accountCombo.setEnabled(false);
			logger.error("Could not populate account combo", e);
		}
	}
