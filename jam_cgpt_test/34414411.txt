		RenamePropertyThread.RenamePropertySettings attributes = new RenamePropertyThread.RenamePropertySettings();
		attributes.DocumentName = getObjectId()+".xml";
		attributes.PropertyName = propName;
		attributes.PropertyNewName = propNewName;
		attributes.SourcePath = path;
		attributes.UserId = userId;
		RenamePropertyThread rpt = _containerEditor.renameProperty(attributes);
		try{
			rpt.start();
			rpt.join();
		}
		catch(InterruptedException ie){
			ie.printStackTrace(System.err);
		}
		rpt.cleanThread();
	}
