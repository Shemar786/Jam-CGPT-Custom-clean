TDAT:   public void fileExit(ActionEvent e) {

    try {
      getDesktop().closeAllWindow();
      appProps.savetoFile( (String) appProps.get(wpPropertyInterface.inifile));
    }
    catch (wpException we) {
      statusBar.setText(we.getMessage());
    }
    // Garbage Collect and bye bye
    System.gc();
    System.exit(0);
  }
