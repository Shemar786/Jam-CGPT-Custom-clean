		if (freeField == null) {
			return false;
		} else {
			if (fields[freeField.x][freeField.y] != null){
				fields[freeField.x][freeField.y].put(item);
			} else {
				fields[freeField.x][freeField.y] = new InventoryField(new InventoryItem(item));
			}
			return true;
		}
		
	}
