TDAT:     private String scanIdentifier(byte[] input, int offset, int end) {
        int begin = offset;

        while ((offset < end) && (this.isIdentifierChar(input[offset]))) {
            offset++;
        }

        if ((offset == end) || (offset == begin)) {
            return null;
        } else {
            return new String(input, begin, offset - begin);
        }
    }
