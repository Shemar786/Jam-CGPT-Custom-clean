		BigDecimal total = new BigDecimal(0);
		IPaymentDAO dao = DAOFactory.getCurrentFactory().getPaymentDAO();
		List<Payment> payments = dao.findByAccount(this);
		for (Payment payment : payments) {
			total = total.add(payment.getPaymentAmount(getCurrency()));
		}
		if (includeChildren) {
			for (Account account : getChildren()) {
				total = total.add(account.getTotal(includeChildren));
			}
		}
		return total;
	}
