TDAT: 	protected String rawReadLine() throws IOException {
		String string = reader.readLine();
		// at end of file, string == null
		// skip comment lines and empty lines
		while (string != null
				&& (string.equals("") || string.matches("[#;].*"))) {
			string = reader.readLine();
		}
		if (string == null)
			throw new EOFException("end of file reached");
		if (string.equals("%"))
			return null;
		if (string.equals(" %"))
			return "";
		return string;
	}
