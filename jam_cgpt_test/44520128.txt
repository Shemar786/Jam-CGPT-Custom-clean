      Iterator f;
      boolean same = false;

      for (f = visibleEdges.iterator();f.hasNext();){
	  Edge edge = (Edge)f.next();
	  if (Edge.sameEdge(e, edge)){
	      same = true;
	      e = edge;
	      break;
	  }
      }
      if (same)
	  visibleEdges.remove(e);
      else
	  visibleEdges.add(e);
  }
