TDAT:     private void connectViaTelnet(){
    	if(mTelnetServer==null){
    		mTelnetServer = new TelnetServer(this);
    		mTelnetServer.connect();
    	}else{
    		addMessage("Closing telnet port " + mTelnetServer.port);
    		mTelnetServer.disconnect();
    		mTelnetServer = new TelnetServer(this);
    		mTelnetServer.connect();
    	}
    }
