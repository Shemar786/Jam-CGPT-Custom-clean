TDAT:   private Point convertParentToOwn(Point point) {
    if (this.getParent() != null) {
      point.x -= this.getLocationOnScreen().x
        - this.getParent().getLocationOnScreen().x;
      point.y -= this.getLocationOnScreen().y;
