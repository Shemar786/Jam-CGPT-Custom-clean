		String logItem = null;

		else if ( subSystemLoggerName.contains("r3") )
			logItem = R3_LOG_FILE;
		else if ( subSystemLoggerName.contains("r4") )
			logItem = R4_LOG_FILE;
		else if ( subSystemLoggerName.contains("network") )
			logItem = NETWORK_LOG_FILE;
		else
			logItem = PLATFORM_LOG_FILE;

		return logItem;
	}
